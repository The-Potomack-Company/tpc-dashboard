    import { useState, type FormEvent } from 'react';
    import { useNavigate } from 'react-router';
    import { useAuthStore } from '../stores/authStore';

    export function LoginPage() {
      const [email, setEmail] = useState('');
      const [password, setPassword] = useState('');
      const [error, setError] = useState&lt;string | null&gt;(null);
      const [submitting, setSubmitting] = useState(false);

        setError(null);
        setSubmitting(true);

          navigate('/', { replace: true });

          &lt;div className="w-full max-w-sm mx-4"&gt;

              TPC Dashboard

              Auction analytics for The Potomack Company

              &lt;div className="mb-4"&gt;
                &lt;label htmlFor="email" className="block text-sm font-semibold text-gray-700 dark:text-gray-300 mb-1"&gt;
                  Email
                &lt;/label&gt;
                &lt;input
                  id="email"
                  type="email"
                  placeholder="you@example.com"
                  value={email}
                  onChange={(e) =&gt; setEmail(e.target.value)}
                  required
                  autoFocus
                  className="w-full min-h-12 px-4 py-3 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-800 text-gray-900 dark:text-gray-100 focus:ring-2 focus:ring-accent focus:border-accent outline-none"

    --- Step 5: Write `src/components/ProtectedRoute.tsx`
    ```typescript
    import { Navigate, Outlet } from 'react-router';
    import { useAuthStore } from '../stores/authStore';
    import { AccessDenied } from './AccessDenied';

    export function ProtectedRoute() {
      const session = useAuthStore((s) =&gt; s.session);
      const loading = useAuthStore((s) =&gt; s.loading);
      const profile = useAuthStore((s) =&gt; s.profile);
      const profileLoading = useAuthStore((s) =&gt; s.profileLoading);
      const isAdmin = useAuthStore((s) =&gt; s.isAdmin);

      // Stage 1: session resolving
      if (loading) {
        return (
          &lt;div
            data-testid="auth-loading"
            aria-label="Checking your session"
            className="flex items-center justify-center h-dvh bg-white dark:bg-gray-900"
          &gt;
            &lt;div className="animate-spin rounded-full h-8 w-8 border-b-2 border-accent" /&gt;
          &lt;/div&gt;
        );
      }

      // Stage 2: not signed in
      if (!session) {
        return &lt;Navigate to="/login" replace /&gt;;
      }

      // Stage 3: profile still loading (prevents flash)
      if (profileLoading || profile === null) {
        return (
          &lt;div
            data-testid="profile-loading"
            aria-label="Checking your session"
            className="flex items-center justify-center h-dvh bg-white dark:bg-gray-900"
          &gt;
            &lt;div className="animate-spin rounded-full h-8 w-8 border-b-2 border-accent" /&gt;
          &lt;/div&gt;
        );
      }

      // Stage 4: not admin
      if (!isAdmin) {
        return &lt;AccessDenied /&gt;;
      }

      // Stage 5: admin — render nested routes
      return &lt;Outlet /&gt;;
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Unauthenticated browser → /login route | Public form; only exposes "enter credentials" |
| /login form → Supabase Auth API | Credentials cross over TLS; rate-limited server-side |
| Authenticated browser → / route | Must be admin; double-gated by ProtectedRoute (client) + RLS (server) |
| Authenticated browser → sign-out | Clears local session; other tabs unaffected per `scope: 'local'` |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-03 | Information Disclosure | Flash-of-protected-content for non-admin user | mitigate | Three-stage ProtectedRoute in Task 1: loading → profile-loading → isAdmin gate. Test in Task 2 asserts non-admin sees AccessDenied, never dashboard-content. |
| T-04-04 | Elevation of Privilege | Client-side role spoof via devtools setting isAdmin=true | mitigate | Client check is UX only. RLS (Plan 02) is authoritative — spoofed admin JWT still returns 0 rows. Defense in depth. |
| T-04-05 | Information Disclosure | User enumeration via distinct error messages | mitigate | Login error surfaces Supabase's generic "Invalid login credentials" OR our "Incorrect email or password. Try again." fallback. Never "user not found" vs "wrong password". |
| T-04-a11y | Information Disclosure (a11y) | Error not announced to screen readers | mitigate | `role="alert"` on every error `<p>` (Login, AccessDenied). Tests assert `getByRole('alert')`. |
