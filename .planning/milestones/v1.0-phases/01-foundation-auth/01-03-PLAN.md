    import { createClient, type SupabaseClient } from '@supabase/supabase-js';
    import type { Database } from '../db/database.types';

    let _client: SupabaseClient&lt;Database&gt; | null = null;

    export const supabase: SupabaseClient&lt;Database&gt; = new Proxy({} as SupabaseClient&lt;Database&gt;, {
      get(_target, prop) {
        if (!_client) {
          const supabaseUrl = import.meta.env.VITE_SUPABASE_URL;
          const supabaseAnonKey = import.meta.env.VITE_SUPABASE_ANON_KEY;
          if (!supabaseUrl) {
            throw new Error('VITE_SUPABASE_URL is not set. Add it to .env.local');
          }
          if (!supabaseAnonKey) {
            throw new Error('VITE_SUPABASE_ANON_KEY is not set. Add it to .env.local');
          }
          _client = createClient&lt;Database&gt;(supabaseUrl, supabaseAnonKey);
        }
        return (_client as unknown as Record&lt;string | symbol, unknown&gt;)[prop];
      },
    });

    --- Step 2: Write `src/stores/authStore.ts`
    ```typescript
    import { create } from 'zustand';
    import { supabase } from '../lib/supabase';
    import type { Session, User } from '@supabase/supabase-js';
    import type { Database } from '../db/database.types';

    type Profile = Database['public']['Tables']['profiles']['Row'];

    interface AuthState {
      session: Session | null;
      user: User | null;
      profile: Profile | null;
      isAdmin: boolean;
      loading: boolean;
      profileLoading: boolean;
      initialize: () =&gt; () =&gt; void;
      signIn: (email: string, password: string) =&gt; Promise&lt;{ error: Error | null }&gt;;
      signOut: () =&gt; Promise&lt;void&gt;;
    }

    export const useAuthStore = create&lt;AuthState&gt;()((set) =&gt; ({
      session: null,
      user: null,
      profile: null,
      isAdmin: false,
      loading: true,
      profileLoading: false,

      initialize: () =&gt; {
        const { data: { subscription } } = supabase.auth.onAuthStateChange(
          async (_event, session) =&gt; {
            set({ session, user: session?.user ?? null, loading: false });

            if (session?.user) {
              set({ profileLoading: true });
              const { data } = await supabase
                .from('profiles')
                .select('*')
                .eq('id', session.user.id)
                .single();
              set({
                profile: data ?? null,
                isAdmin: data?.role === 'admin',
                profileLoading: false,
              });
            } else {
              set({ profile: null, isAdmin: false, profileLoading: false });
            }
          }
        );
        return () =&gt; subscription.unsubscribe();
      },

      signIn: async (email, password) =&gt; {
        const { error } = await supabase.auth.signInWithPassword({ email, password });
        return { error };
      },

      signOut: async () =&gt; {
        await supabase.auth.signOut({ scope: 'local' });
      },
    }));
    ```
