    ```ts
    // Phase 5 Plan 05-06 — heat map quintile bucketing + sorted dept codes.
    // Contract: .planning/phases/05-trend-analysis/05-UI-SPEC.md § Heat-map color ramp.
    // 5-bucket min-max ramp using Tailwind's blue-100/300/500/700/900 tiers.
    // Sorting uses ASCII compare — all dept codes are uppercase letters so
    // lexicographic order === alphabetical order.

    const DEPT_CODES_RAW = [
      'AMER', 'ASD', 'ASN', 'ASNP', 'BKS', 'CER', 'CLK', 'DEC', 'DRW', 'ENT',
      'FRN', 'GEN', 'GLS', 'MAP', 'MDF', 'MUS', 'PER', 'PND', 'PNT', 'SIL',
      'SPT', 'TXTL',
    ];
    export const SORTED_DEPT_CODES: readonly string[] = Object.freeze(
      [...DEPT_CODES_RAW].sort(),
    );

    const BUCKET_CLASSES = [
      'bg-blue-100 dark:bg-blue-900/40',
      'bg-blue-300 dark:bg-blue-800/60',
      'bg-blue-500 dark:bg-blue-700/80',
      'bg-blue-700 dark:bg-blue-600',
      'bg-blue-900 dark:bg-blue-500',
    ] as const;

    export function bucketClassFor(value: number, min: number, max: number): string {
      if (min === max) return BUCKET_CLASSES[4];
      const n = (value - min) / (max - min);
      if (n < 0.2) return BUCKET_CLASSES[0];
      if (n < 0.4) return BUCKET_CLASSES[1];
      if (n < 0.6) return BUCKET_CLASSES[2];
      if (n < 0.8) return BUCKET_CLASSES[3];
      return BUCKET_CLASSES[4];
    }

    export const NO_DATA_CELL_CLASS = 'bg-gray-50 dark:bg-gray-800';
    export const NO_DATA_CELL_STYLE: Record<string, string> = {
      backgroundImage:
        'repeating-linear-gradient(45deg, transparent, transparent 4px, rgba(156, 163, 175, 0.3) 4px, rgba(156, 163, 175, 0.3) 5px)',
    };
    ```

    Props: `{ range: Range; metric: HeatMapMetric }`.

    Data pipeline:
    1. Call `useDepartmentGrid(range)`.
    2. Filter rows: `sale_date != null` (sales without dates don't fit on a column axis).
    3. Sort rows by `sale_date` ASC. (Hook already sorts ASC, but re-sort defensively after filtering.)
    4. Build a cell-lookup map: `cells: Record<\`${deptCode}|${saleNumber}\`, number | null>` where the value depends on `metric`:
       - `sell_through`: `dept.sell_through_pct / 100` (convert 0..100 to 0..1 ratio; null stays null).
       - `revenue_share`: compute total sale revenue = sum of `d.revenue` for that sale's dept rows (skipping null); `dept.revenue / totalSaleRevenue` if totalSaleRevenue > 0; null if revenue is null or totalSaleRevenue is 0.
    5. Compute `min` and `max` across all non-null cell values.
    6. Derive `metricLabel = metric === 'sell_through' ? 'Sell-through' : 'Revenue share'`.

    Rendering states:
    - Pending → `<ChartSkeleton height="lg" />`.
    - Error → `<ErrorState heading="Couldn't load this chart" body="Something went wrong fetching department data for the selected range. Retry below, or try a different range." onRetry={refetch} />`.
    - Empty (no filtered rows) → `<EmptyState heading="No department data in this range"><p>Try expanding the date filter, or switch the metric above.</p></EmptyState>`.
    - Success → grid.

    Grid structure (UI-SPEC lines 656-711):

    ```tsx
              {s.sale_number}
            </div>
          ))}

          {/* Rows 2..23: dept code + 22 cells */}
          {SORTED_DEPT_CODES.map(code => (

                   role="rowheader">
                {code}
              </div>
              {sales.map(s => {
                const v = cells[`${code}|${s.sale_number}`] ?? null;
    ```
