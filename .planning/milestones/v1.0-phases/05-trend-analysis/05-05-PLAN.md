    - `total_sold_value < low_estimate` → count into `below_lots_sum`.
    - `low_estimate <= total_sold_value <= high_estimate` → `within_lots_sum`.
    - `total_sold_value > high_estimate` → `above_lots_sum`.

    Output:
    - If `saleLotsSold == null || saleLotsSold === 0` → return `null` (no denominator, no meaningful share).
    - If the three sums together are all zero (every dept skipped) → return `null`.
    - Otherwise: `{ below: below_lots_sum / saleLotsSold, within: within_lots_sum / saleLotsSold, above: above_lots_sum / saleLotsSold }`.

    NOTE: the three values may NOT sum to exactly 1.0 when some departments are skipped due to null data, because the denominator is sale-level lots_sold (which includes those nulls). This is intentional — the stacked area at 100% mode will auto-normalize what we pass, but the raw ratios preserve the "missing data" fact. Decision: emit the raw ratios; let Recharts' `stackOffset="expand"` renormalize for visual stacking. Tooltip shows raw per-band percentages.

    ```ts
    // Phase 5 Plan 05-05 — TRND-05 estimate accuracy derivation.
    // Contract: .planning/phases/05-trend-analysis/05-CONTEXT.md § TRND-05.
    // Per-sale classification: sum lots_sold of dept rows grouped by whether
    // total_sold_value is below / within / above [low_estimate, high_estimate],
    // expressed as share of sale-level lots_sold.
    // Returns null when no meaningful computation is possible.

    export interface AccuracyBands {
      below: number;
      within: number;
      above: number;
    }

    interface DeptInput {
      total_sold_value: number | null;
      low_estimate: number | null;
      high_estimate: number | null;
      lots_sold: number | null;
    }

    export function computeAccuracyBands(
      depts: ReadonlyArray<DeptInput>,
      saleLotsSold: number | null,
    ): AccuracyBands | null {
      if (saleLotsSold == null || saleLotsSold === 0) return null;

      let below = 0;
      let within = 0;
      let above = 0;
      let any = false;

      for (const d of depts) {
        if (d.total_sold_value == null || d.low_estimate == null ||
            d.high_estimate == null || d.lots_sold == null) continue;
        any = true;
        if (d.total_sold_value < d.low_estimate) below += d.lots_sold;
        else if (d.total_sold_value > d.high_estimate) above += d.lots_sold;
        else within += d.lots_sold;
      }

      if (!any) return null;
      return {
        below: below / saleLotsSold,
        within: within / saleLotsSold,
        above: above / saleLotsSold,
      };
    }
    ```
