       interface SortIndicatorProps {
         state: 'asc' | 'desc' | false;

       export function SortIndicator({ state }: SortIndicatorProps) {
         const activeClass = 'text-accent';
         const inactiveClass = 'text-gray-400 dark:text-gray-500';
         const className = `w-4 h-4 ${state === false ? inactiveClass : activeClass}`;

         if (state === 'asc') {
           // chevron-up outline, stroke-width 1.5
           return (
             <svg xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24" strokeWidth={1.5} stroke="currentColor" className={className} aria-hidden="true">
               <path strokeLinecap="round" strokeLinejoin="round" d="m4.5 15.75 7.5-7.5 7.5 7.5" />
             </svg>
           );
         }
         if (state === 'desc') {
           // chevron-down
           return (
             <svg xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24" strokeWidth={1.5} stroke="currentColor" className={className} aria-hidden="true">
               <path strokeLinecap="round" strokeLinejoin="round" d="m19.5 8.25-7.5 7.5-7.5-7.5" />
             </svg>
           );
         }
         // chevron-up-down (inactive)
         return (
           <svg xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24" strokeWidth={1.5} stroke="currentColor" className={className} aria-hidden="true">
             <path strokeLinecap="round" strokeLinejoin="round" d="M8.25 15 12 18.75 15.75 15m-7.5-6L12 5.25 15.75 9" />
