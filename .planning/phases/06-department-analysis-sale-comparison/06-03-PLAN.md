    // Stable across re-renders by memoizing a Map.
    const colorForCode = useCallback((code: string): string => {
      const idx = chipSelectedDepts.indexOf(code);
      if (idx >= 0) return CHART_PALETTE[idx % CHART_PALETTE.length];

      const map: Record<string, string | null> = {};
