    - Returns a new array of the same length as `values`.
    - For index `i < window - 1`: returns `null` (insufficient history — "rolling-3 mean at position 0 or 1 is undefined").
    - For index `i >= window - 1`: returns the arithmetic mean of `values[i - window + 1 .. i]` if all those values are non-null; returns `null` if any in the window is null.
    - Does NOT mutate the input.
    - Throws if `window < 1` or `window > values.length && values.length > 0` → return all nulls rather than throw (graceful).

    Actually — specify the non-throw path: if `window > values.length`, return an all-null array of the same length. If `window < 1`, throw `new Error('window must be >= 1')`.

    ```ts
    // Phase 5 — rolling mean helper for TRND-01 / TRND-02 trend overlays.
    // CONTRACT: .planning/phases/05-trend-analysis/05-CONTEXT.md § Required Charts.
    // Window = 3 in Phase 5; the function accepts any window >= 1 for future
    // configurability (TRND-CONTEXT § Claude's Discretion — NOT user-configurable yet).

    export function computeRollingMean(
      values: ReadonlyArray<number | null>,
      window: number,
    ): Array<number | null> {
      if (window < 1) throw new Error('window must be >= 1');
      const n = values.length;
      const out: Array<number | null> = new Array(n).fill(null);
      if (window > n) return out;
      for (let i = window - 1; i < n; i++) {
        let sum = 0;
        let anyNull = false;
        for (let j = i - window + 1; j <= i; j++) {
          const v = values[j];
          if (v === null || v === undefined) { anyNull = true; break; }
          sum += v;
        }
        out[i] = anyNull ? null : sum / window;
      }
      return out;
    }
    ```
