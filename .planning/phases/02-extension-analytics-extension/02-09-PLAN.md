```typescript
// Build a chainable mock that resolves to fixture data per query type.
function makeSupabaseStub(opts: { gateRows?: unknown[]; rpcData?: unknown[]; selectData?: unknown[] }) {
  const rpcMock = vi.fn().mockResolvedValue({ data: opts.rpcData ?? [], error: null });
  function makeChain(finalData: unknown[]) {
    const chain: Record<string, unknown> = {};
    const passthrough = ['select','eq','not','in','gte','lte','order','limit'] as const;
    for (const m of passthrough) (chain as Record<string, () => unknown>)[m] = () => chain;
    (chain as { then: (cb: (v: unknown) => unknown) => Promise<unknown> }).then = (cb) =>
      Promise.resolve({ data: finalData, error: null }).then(cb);
    return chain;
  }
  return {
    rpc: rpcMock,
    from: vi.fn().mockImplementation((table: string) => {
      if (table === 'analytics_events') {
        // Could differentiate based on test setup — for the smoke we use one fixture for the gate probe + selects
        return makeChain(opts.gateRows ?? opts.selectData ?? []);
      }
      return makeChain([]);
    }),
  };
}
```

Implementation notes:
- The `vi.hoisted` pattern lets us swap the Supabase stub per test without re-mocking the module. If your project's Vitest version doesn't support `vi.hoisted` cleanly, fall back to a module-level mutable factory + `vi.resetModules` between tests.
- The chain mock is necessarily approximate; if a future query adds new chained methods, extend the `passthrough` array. The smoke test is OK with low fidelity here — its job is "page composes without crashes," not "every Supabase call has correct args" (which is Plan 02-03 Task 1's job).
- Console.error spy in Test 1 catches unexpected React warnings (key prop warnings, hook violations, prop-type errors). Real `console.error` calls during the test trip the assertion — pin the page to console-clean mounting.

```typescript
it('filter change re-fetches all charts (EXT-07 integration)', async () => {
  // Track supabase calls so we can prove a refetch happened with the new filter.
  const allSupabaseCalls: Array<{ method: string; args: unknown[] }> = [];

  function makeRecordingChain(finalData: unknown[]) {
    const chain: Record<string, unknown> = {};
    const passthrough = ['select','eq','not','in','gte','lte','order','limit'] as const;
    for (const m of passthrough) {
      (chain as Record<string, (...a: unknown[]) => unknown>)[m] = (...args: unknown[]) => {
        allSupabaseCalls.push({ method: m, args });
        return chain;
      };
    }
    (chain as { then: (cb: (v: unknown) => unknown) => Promise<unknown> }).then = (cb) =>
      Promise.resolve({ data: finalData, error: null }).then(cb);
    return chain;
  }

  // Stub: gate has rows; option list has user@x.com so UserMultiSelect can show it
  supabaseStub.current = {
    rpc: vi.fn().mockImplementation((_name: string, args: Record<string, unknown>) => {
      allSupabaseCalls.push({ method: 'rpc', args: [_name, args] });
      return Promise.resolve({ data: [], error: null });
    }),
    from: vi.fn().mockImplementation(() => makeRecordingChain([
      { id: '1', user_email_label: 'user@x.com', extension_version: '2.0.1' },
    ])),
  };

  // Mount with ?range=7d
  const client = new QueryClient({ defaultOptions: { queries: { retry: false, gcTime: 0 } } });
  render(
    <MemoryRouter initialEntries={['/extension?range=7d']}>
      <QueryClientProvider client={client}>
        <ExtensionPage />
      </QueryClientProvider>
    </MemoryRouter>,
  );

  // Wait for the gate probe + initial chart fetches to settle
  await waitFor(() => {
    expect(screen.queryByTestId('extension-page-loading')).not.toBeInTheDocument();
  });

  // Snapshot calls so we can compare BEFORE vs AFTER filter change
  const callsBefore = allSupabaseCalls.length;

  // Real UserMultiSelect is mounted — open the popover and toggle a user
  await userEvent.click(screen.getByRole('button', { name: 'Filter by user email' }));
  await userEvent.click(screen.getByLabelText(/user@x\.com/));

  // Close the popover (Escape or click elsewhere)
  await userEvent.keyboard('{Escape}');

  // Await the next fetch tick — TanStack should re-fire all chart hooks with the new queryKey
  await waitFor(() => {
    expect(allSupabaseCalls.length).toBeGreaterThan(callsBefore);
  });

  // Verify at least ONE new call carries the user filter
  // Either as `.in('user_email', ['user@x.com'])` (raw select) OR `p_users: ['user@x.com']` (RPC)
  const newCalls = allSupabaseCalls.slice(callsBefore);
  const hasUserFilter = newCalls.some((c) => {
    if (c.method === 'in' && c.args[0] === 'user_email' && Array.isArray(c.args[1])) {
      return (c.args[1] as string[]).includes('user@x.com');
    }
    if (c.method === 'rpc') {
      const rpcArgs = c.args[1] as { p_users?: string[] } | undefined;
      return Array.isArray(rpcArgs?.p_users) && rpcArgs.p_users.includes('user@x.com');
    }
    return false;
  });
  expect(hasUserFilter).toBe(true);
});
```

- The recording chain mock captures every `.in(...)` and `.rpc(...)` call; the assertion accepts either RPC (`p_users` arg) or raw select (`.in('user_email', [...])`) as proof — both shapes are valid filter propagation.
- This test mounts the REAL UserMultiSelect (no `vi.mock('../components/UserMultiSelect')`) — that's intentional. The 3 prior tests can stub UserMultiSelect to avoid noise, but Test 4 specifically exercises the filter-change path through the composed page.

If smoke proves flaky due to async timing in mocked TanStack Query (see PATTERNS.md notes about microtask ordering with `vi.useFakeTimers`), DO NOT use fake timers in this file. Real timers + `await waitFor(...)` is sufficient.
