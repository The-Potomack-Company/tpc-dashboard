       ```
         const progress = `[ ${i+1}/${files.length} ] ${saleNumber.padEnd(8)}`;
         const parsed = await parsePdf(join(source, file));
         if (parsed.status === 'empty') {
           summary.skipped_empty++;
           io.out(`${progress} → skipped: empty_placeholder`);
           continue;
         }
         if (parsed.status === 'failed') {
           summary.failed++;
           summary.failures.push({ sale_number: saleNumber, error: parsed.error });
           io.out(`${progress} → failed: ${parsed.error}`);
           continue;
         }
         if (dryRun) {
           summary.dry_run_ok++;
           io.out(`${progress} → (dry-run) ok: ${parsed.departments.length} depts`);
           continue;
         }
         const result = await importSale(parsed.sale, parsed.departments, { toleranceCents: Math.round(toleranceDollars * 100) });
         if (result.ok === true) {
           summary.inserted++;
           if (result.validationWarning) {
             summary.validation_warnings++;
             io.out(`${progress} → inserted (WARN: ${result.mismatches?.join('; ')})`);
           } else {
             io.out(`${progress} → inserted`);
           }
         } else if (result.reason === 'duplicate') {
           summary.skipped_duplicate++;
           io.out(`${progress} → skipped: duplicate`);
         } else {
           summary.failed++;
           summary.failures.push({ sale_number: saleNumber, error: result.error });
           io.out(`${progress} → failed: ${result.error}`);
         }
       }
       ```

| Boundary | Description |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05 | Tampering (wrong-DB target) | CLI banner + 3-second delay | mitigate | Confirmation banner prints source dir + target Supabase URL + file count + mode before any DB write. 3-second pause lets operator Ctrl+C abort. Acceptance criterion grep-verifies banner strings. |

Error messages from RPC are logged but those never contain the key (they contain SQL error messages). |
