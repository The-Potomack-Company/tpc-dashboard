      const INLINE_FIELD_RE = /^(?<label>[^:]+?):\s+(?<value>\S.*)$/;
      const HEADER_RE = /^Auction Profile for Sale\s+(?<sale>\S+),\s+(?<date>[A-Z][a-z]+\s+\d{1,2},\s+\d{4})\s*$/;
      const LEFT_COLUMN_LABELS = [ /* 12 labels from RESEARCH.md Pattern 3 lines 260-273 */ ];

      export function parseSalePage(pageText: string, opts: { sourcePdfPath: string }): Omit<SaleRecord, 'imported_at'> {
        const lines = pageText.split('\n').map(l => l.replace(/\r$/, ''));
        const tabFields: Record<string, string> = {};
        const inlineFields: Record<string, string> = {};
        for (const line of lines) {
          const t = TAB_FIELD_RE.exec(line);
          if (t?.groups) tabFields[t.groups.label.trim()] = t.groups.value.trim();
          const i = INLINE_FIELD_RE.exec(line);
          if (i?.groups && !t) inlineFields[i.groups.label.trim()] = i.groups.value.trim();
        }
        const headerLine = lines.find(l => HEADER_RE.test(l));
        const hm = headerLine ? HEADER_RE.exec(headerLine) : null;
        const sale_number = hm?.groups?.sale ?? '';
        const sale_date = hm?.groups?.date ? new Date(hm.groups.date).toISOString().slice(0, 10) : null;
        const headerIdx = lines.findIndex(l => HEADER_RE.test(l));
        const title = (lines[headerIdx + 1] ?? '').trim();
        const leftCol = extractLeftColumnBlock(lines, LEFT_COLUMN_LABELS);
        const lotsAuctioned = parseAuctionedLots(leftCol['Auctioned Lots'] ?? inlineFields['Auctioned Lots'] ?? '');
        const lotsSold = parseLotsSold(leftCol['Lots Sold'] ?? '');
        const totalEstimate = parseMoneyRange(leftCol['Total Estimate'] ?? '');
        return {
          sale_number,
          title: title || `Sale ${sale_number}`,
          sale_date,
          lots_auctioned: lotsAuctioned.count,
          lots_sold: lotsSold.count,
          lots_unsold: (lotsAuctioned.count != null && lotsSold.count != null) ? lotsAuctioned.count - lotsSold.count : null,
          total_sold_value: parseMoney(tabFields['Total Sold Value'] ?? leftCol['Total Sold Value']),
          total_unsold_value: parseMoney(leftCol['Total Unsold Value']),
          total_low_estimate: totalEstimate.low,
          total_high_estimate: totalEstimate.high,
          total_reserves: parseMoney(inlineFields['Total Reserve']),
          hammer_total: parseMoney(tabFields['Total Hammer Paid']),
          buyer_premium: parseMoney(tabFields['Premium']),
          seller_commission: parseMoney(tabFields['Commission']),
          insurance: parseMoney(tabFields['Insurance']),
          lot_charges: parseMoney(tabFields['Lot Charges']),
          referral_fees: parseMoney(tabFields['Referral Fees']),
          net_revenue: parseMoney(tabFields['Total Net Revenue']),
          registered_bidders: parseCount(leftCol['Registered Bidders']),
          winning_buyers: parseCount(leftCol['Number of Buyers']),
          payment_status: derivePaymentStatus(inlineFields),
          source_pdf_path: opts.sourcePdfPath,
          validation_warning: false,
        };
      }

      import { readFile } from 'node:fs/promises';
      import { PDFParse } from 'pdf-parse';
      import { parseSalePage } from './parsers/sale-page.js';
      import { parseDepartmentPage } from './parsers/department-page.js';
      import { SaleRecordSchema, SaleDepartmentRecordSchema, type SaleRecord, type SaleDepartmentRecord } from './schemas.js';

      export type ParseResult =
        | { status: 'ok'; sale: SaleRecord; departments: SaleDepartmentRecord[] }
        | { status: 'empty' }
        | { status: 'failed'; error: string };

      export async function parsePdf(filePath: string): Promise<ParseResult> {
        let buffer: Buffer;
        try {
          buffer = await readFile(filePath);
        } catch (err) {
          return { status: 'failed', error: `read: ${err instanceof Error ? err.message : String(err)}` };
        }

        // Fast-path: 1182-byte empty placeholders
        if (buffer.byteLength <= 1200) {
          try {
            const p = new PDFParse({ data: buffer });
            const r = await p.getText();
            if (r.pages.length === 1 && r.pages[0].text.trim().length === 0) {
              return { status: 'empty' };
            }
          } catch {
            return { status: 'empty' };
          }
        }

        let result;
        try {
          const p = new PDFParse({ data: buffer });
          result = await p.getText();
        } catch (err) {
          return { status: 'failed', error: `pdf-parse: ${err instanceof Error ? err.message : String(err)}` };
        }

        if (!result.pages || result.pages.length === 0 || result.pages[0].text.trim().length === 0) {
          return { status: 'empty' };
        }

        try {
          const saleRaw = parseSalePage(result.pages[0].text, { sourcePdfPath: filePath });
          const sale = SaleRecordSchema.parse(saleRaw);
          const departments: SaleDepartmentRecord[] = [];
          for (let i = 1; i < result.pages.length; i++) {
            const deptRaw = parseDepartmentPage(result.pages[i].text);
            if (!deptRaw.code) continue;  // skip pages without a dept header (e.g., footer-only pages)
            const dept = SaleDepartmentRecordSchema.parse(deptRaw);
            departments.push(dept);
          }
          return { status: 'ok', sale, departments };
        } catch (err) {
          return { status: 'failed', error: err instanceof Error ? err.message : String(err) };
        }
      }

    T-03 mitigation: Every pdf-parse call is in try/catch. A malformed PDF cannot crash the importer; it becomes { status: 'failed' } and the CLI reports it and moves on.
